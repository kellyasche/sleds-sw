The dataset above contains each county along with their three year rolling average of unemployment rate.

Now, we need to join it up with the master dataset by high school graduation year and county of high school.


master.11 <- read_csv("Data/SLEDS/Masters/Master-11.csv")

master.12 <- master.11 %>%
  left_join(unemployment.original, by = c("hs.grad.year", "countyfp"))

unemployment.master <- read_csv("Data/SLEDS/Masters/Master-1.csv") %>%
  select(PersonID, grad.year, countyfp) %>%
  left_join(unemployment.original, by = c("grad.year" = "hs.grad.year", "countyfp")) %>%
  select(-countyfp, -grad.year)

write_csv(unemployment.master, "Data/SLEDS/Masters/unemployment_master.csv")
